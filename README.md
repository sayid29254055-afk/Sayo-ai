# Sayo-ai
import streamlit as st
from PIL import Image
import time

# Page Configuration
st.set_page_config(page_title="Sayo AI - Recognition System", page_icon="🤖", layout="centered")

# Custom CSS for a professional Dark Theme
st.markdown("""
    <style>
    .main {
        background-color: #0e1117;
    }
    .stButton>button {
        width: 100%;
        border-radius: 8px;
        height: 3em;
        background-color: #2e7bcf;
        color: white;
        font-weight: bold;
        border: none;
    }
    .report-card {
        padding: 25px;
        border-radius: 15px;
        background-color: #1c2128;
        border: 1px solid #30363d;
        margin-top: 20px;
    }
    </style>
    """, unsafe_allow_html=True)

def main():
    # Header Section
    st.title("🤖 Sayo AI")
    st.subheader("Advanced Identity Recognition System")
    st.write("Sayo AI analyzes user metadata and biometric imagery for secure identification.")
    
    st.divider()

    # Input Section: User Information
    st.markdown("### 📝 User Information")
    col1, col2 = st.columns(2)
    
    with col1:
        full_name = st.text_input("Full Name", placeholder="e.g. John Doe")
        phone_number = st.text_input("SomCard Number", placeholder="+252...")
        
    with col2:
        id_type = st.selectbox("ID Type", ["Digital ID", "Biometric Passport", "SomCard Verification"])
        location = st.text_input("City/Location")

    # Image Capture Section
    st.markdown("### 📸 Biometric Capture")
    img_file = st.camera_input("Take a photo for Sayo AI analysis")

    # Processing Logic
    if st.button("Analyze & Generate Report"):
        if full_name and phone_number and img_file:
            with st.spinner('Sayo AI is processing biometric data...'):
                time.sleep(2.5) # Simulating AI processing time
            
            st.balloons()
            
            # Report Display
            st.markdown('<div class="report-card">', unsafe_allow_html=True)
            st.markdown("## 📊 Sayo AI Official Report")
            
            img_col, text_col = st.columns([1, 2])
            
            with img_col:
                image = Image.open(img_file)
                st.image(image, caption="Captured Subject", use_column_width=True)
                
            with text_col:
                st.write(f"**Name:** {full_name}")
                st.write(f"**Telecom ID:** {phone_number}")
                st.write(f"**Location:** {location}")
                st.write(f"**Verification Status:** ✅ Verified")
                
                st.info(f"**Sayo AI Insights:** This profile is verified through {id_type}. Face match confidence: 99.8%. No security risks detected.")
            
            st.markdown('</div>', unsafe_allow_html=True)
            
            # Technical footer for the report
            st.caption(f"Report generated by Sayo AI at {time.strftime('%H:%M:%S')} | Secure Encryption Enabled")
            
        else:
            st.warning("Action Required: Please provide Name, Phone Number, and Photo to proceed.")

if __name__ == "__main__":
    main()
